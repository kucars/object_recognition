cmake_minimum_required(VERSION 2.8.3)
project(objectrecognition)

## Find catkin macros and libraries
## if COMPONENTS list like find_package(catkin REQUIRED COMPONENTS xyz)
## is used, also find other catkin packages
find_package(catkin REQUIRED COMPONENTS
  roscpp
  pcl_ros
  rospy
  std_msgs
  bottomup_msgs
  rosconsole
  tabletop_object_segmentation_online
  tf
  pcl_conversions
  perception_msgs
)

## System dependencies are found with CMake's conventions
find_package(Boost REQUIRED COMPONENTS system)
find_package(PCL REQUIRED)
find_package(Eigen REQUIRED)
find_package(VTK REQUIRED)

include_directories(include   
  ${Boost_INCLUDE_DIRS}
  ${catkin_INCLUDE_DIRS}
  ${Eigen_INCLUDE_DIRS}
  ${PCL_INCLUDE_DIRS}
  ${VTK_INCLUDE_DIRS}
)

## Add link directories
link_directories(
  ${Boost_LIBRARY_DIRS}
  ${Eigen_LIBRARY_DIRS}
  ${PCL_LIBRARY_DIRS}
)

catkin_package(
  INCLUDE_DIRS include
  LIBRARIES
    objectrecognition_sv 
  CATKIN_DEPENDS
    roscpp
    bottomup_msgs
    rosconsole
    #tabletop_object_segmentation_online
    tf
    pcl_conversions
    pcl_ros
    perception_msgs
  DEPENDS
    #Boost
    Eigen
)

add_library(objectrecognition_sv 
	src/${PROJECT_NAME}/point_pair.cpp  
	src/${PROJECT_NAME}/object_model.cpp
	src/${PROJECT_NAME}/pose_estimation.cpp
	src/${PROJECT_NAME}_sv/object_model.cpp 
	src/${PROJECT_NAME}_sv/pose_estimation.cpp
	src/${PROJECT_NAME}_sv/point_pair.cpp  
	)
